241 void Graph::deleteNode(Node* _handle)
242 {
243 int idx = getIndex(_handle);
244
246 if(adjacencyList_.size() > 1)
247 {
249 if(adjacencyList_.at(idx)->BackCnnctn() == nullptr)
250 {
253
254 std::vector<double> list;
255 for (const auto& i : adjacencyList_.at(idx)->fwd_node_)
256 {
257 list.push_back(i->backEdgeWeight());
258 }
259
260 auto temp = std::min_element(list.begin(), list.end());
261 auto min_idx = std::distance(list.begin(), temp);
262 auto new_head = adjacencyList_.at(idx)->fwd_node_.at(min_idx);
263
265 for(const auto &iter : adjacencyList_.at(idx)->fwd_node_)
266 {
267 if(iter != new_head)
268 {
269 new_head->fwd_node_.push_back(iter);
270 }
271 }
272 new_head->setBackCnnctn(nullptr);
273 new_head->setBackEdgeWeight(0.0F);
274
276 adjacencyList_.erase(adjacencyList_.begin());
277
279 int temp_head_idx = getIndex(new_head);
280 Node* cpy = adjacencyList_.at(temp_head_idx);
281 adjacencyList_.erase(adjacencyList_.begin() + temp_head_idx);
282 adjacencyList_.insert(adjacencyList_.begin(), cpy);
283
285 temp_head_idx = getIndex(_handle);
286
287 //adjacencyList_.erase(adjacencyList_.begin() + temp_head_idx + 1);
288 }
289 else
290 {
291 auto upstream = adjacencyList_.at(idx)->BackCnnctn();
292 auto upstream_idx = getIndex(upstream);
293
295
296 for(const auto &iter_b : adjacencyList_.at(idx)->fwd_node_)
297 {
298 if(iter_b != nullptr)
299 {
300 adjacencyList_.at(upstream_idx)->fwd_node_.push_back(iter_b);
301 }
302 }
303
306 for(const auto &iter_f : adjacencyList_.at(idx)->fwd_node_)
307 {
308 iter_f->setBackCnnctn(adjacencyList_.at(idx));
309 }
310
312 if(upstream != nullptr)
313 {
315 auto fwd_link = std::remove(upstream->fwd_node_.begin(), upstream->fwd_node_.end(), _handle);
316 upstream->fwd_node_.erase(fwd_link);
317 }
318
320 adjacencyList_.erase(adjacencyList_.begin() + idx);
321 }
322 }

361 void Graph::updateEdge(Node* _src, Node* _fwd)
362 {
363 // Remove _fwd from its current back connection's fwd_node_ list
364 Node* old_back = _fwd->BackCnnctn();
365 if (old_back) {
366 auto& fwd_list = old_back->fwd_node_;
367 auto it = std::remove(fwd_list.begin(), fwd_list.end(), _fwd);
368 fwd_list.erase(it, fwd_list.end());
369 }
370 _src->fwd_node_.push_back(_fwd);
371 _fwd->setBackCnnctn(_src);
372 }
