266 Node* RRT::findNearest(Node *_handle, bool _temporal)
267 {
268 int idx = -1;
269 int i = 0;
270 double temp;
271 double min = DBL_MAX;
272
274 for(const auto &iter : adjacencyList_)
275 {
277 if (std::find(destNodes.begin(), destNodes.end(), iter) != destNodes.end())
278 {
279 i++;
280 continue;
281 }
282
284 bool has_fwd_to_dest = false;
285 for (const auto& fwd : iter->fwd_node_)
286 {
287 if (std::find(destNodes.begin(), destNodes.end(), fwd) != destNodes.end())
288 {
289 has_fwd_to_dest = true;
290 break;
291 }
292 }
293 if (has_fwd_to_dest)
294 {
295 i++;
296 continue;
297 }
298
299 temp = calcDist(_handle, iter, _temporal);
300
302 if(temp < min && _handle != iter)
303 {
304 min = temp;
305 idx = i;
306 }
307 i++;
308 }
309
310 // If no valid nearest node found, return nullptr
311 if (idx == -1)
312 return nullptr;
313
314 return adjacencyList_.at(idx);
315 }

333 {
334 double delta_x = _handle->xCrdnt() - _ref->xCrdnt();
335 double delta_y = _handle->yCrdnt() - _ref->yCrdnt();

357 {

359 double dist = calcDist(_handle, nearest, true);
360 double angle = calcAngle(_handle, nearest);

366 (std::abs(dist) > min_dist_ || std::abs(std::abs(dist) - min_dist_) <= eplison) &&

480 double RRT::calcMengerCurvature(Node* _ref0, Node* _ref1, Node* _ref2)
481 {
483 if (!_ref0 || !_ref1 || !_ref2) {
484 return 0.0;
485 }
486
488
489 double a = calcDist(_ref0, _ref1, false);
490 double b = calcDist(_ref1, _ref2, false);
491 double c = calcDist(_ref2, _ref0, false);
492
493 double area = 0.25F * std::sqrt((a + (b + c)) * (c - (a - b)) * (c + (a - b)) * (a + (b - c)));
494
495 if(area < 1e-10 || a == 0.0F || b == 0.0F || c == 0.0F)
496 {
497 return 0.0F; // Points are collinear or too close together
498 }
499 else
500 {
501 double curvature = (4 * area) / (a * b * c);
502 return curvature;
503 }
504 }
505
506 // Provide the missing member overload so the constructor's call to
507 // RRT::calcMengerCurvature(double,double,double) is defined.
508 double RRT::calcMengerCurvature(double _ab, double _bc, double _ac)
509 {
510 double a = _ab;
511 double b = _bc;
512 double c = _ac;
513
514 // Heron's-like formula for triangle area from sides a,b,c (Menger)
515 double area_term = (a + (b + c)) * (c - (a - b)) * (c + (a - b)) * (a + (b - c));
516 if (area_term <= 0.0) {
517 return 0.0;
518 }
519 double area = 0.25 * std::sqrt(area_term);
520
521 if (area < 1e-10 || a == 0.0 || b == 0.0 || c == 0.0)
522 {
523 return 0.0;
524 }
525 double curvature = (4.0 * area) / (a * b * c);
526 return curvature;
527 }

529 void RRT::constrainCurvature(Node* _handle)
530 {
533 if(_handle->BackCnnctn() || _handle->BackCnnctn()->BackCnnctn())
534 {
535 if (max_kappa_rad_ <= 0) {
536 std::cerr << "Curvature must be positive." << std::endl;
537 max_kappa_rad_ = std::abs(max_kappa_rad_);
538 }
539
541 int curvature_sign = calcAngle(_handle, _handle->BackCnnctn()) >= 0 ? 1 : -1;
542
544 double a_x = _handle->BackCnnctn()->BackCnnctn()->xCrdnt();
545 double a_y = _handle->BackCnnctn()->BackCnnctn()->yCrdnt();
546 double b_x = _handle->BackCnnctn()->xCrdnt();
547 double b_y = _handle->BackCnnctn()->yCrdnt();
548
549 // 1. Calculate the radius (R) from the Menger curvature (k).
550 auto radius = 1.0 / max_kappa_rad_;
551
552 // Calc the distance between points a and b
553 double d = calcDist(_handle->BackCnnctn()->BackCnnctn(), _handle->BackCnnctn(), false);
554
555 // 2. Find the center(s) of the circle.
556 // The a-b midpoint
557 pose_t mid = std::make_tuple((a_x + b_x) / 2.0, (a_y + b_y) / 2.0, 0.0F, 0.0F);
558
559 // The distance from the a-b midpoint to the circle center.
560 double h = std::sqrt(std::pow(radius, 2) - std::pow(d / 2.0, 2));
561
562 // The slope and perpendicular slope of a-b.
563 double slope_ab = (b_y - a_y) / (b_x - a_x);
564 double perp_slope_ab = -1.0 / slope_ab;
565
566 // Handle vertical line a-b (undefined slope).
567 double cx_offset, cy_offset;
568 if (std::isinf(perp_slope_ab)) { // a-b is horizontal
569 cx_offset = 0;
570 cy_offset = h;
571 } else if (std::isinf(slope_ab)) { // a-b is vertical
572 cx_offset = h;
573 cy_offset = 0;
574 } else {
575 double angle = std::atan(perp_slope_ab);
576 cx_offset = h * std::cos(angle);
577 cy_offset = h * std::sin(angle);
578 }
579
580 // Calculate the two possible circle centers.
581 pose_t center1 = std::make_tuple(std::get<0>(mid) + cx_offset, std::get<1>(mid) + cy_offset, 0.0F, 0.0F);
582 pose_t center2 = std::make_tuple(std::get<0>(mid) - cx_offset, std::get<1>(mid) - cy_offset, 0.0F, 0.0F);
583
584 // 3. Solve for a third point (c).
585 // The third point is any point on the circle. We can choose one, for example,
586 // by rotating point 'a' around the center by some angle.
587 // This example finds a point on the circle with the same y-coordinate as the center.
588
589 // Solution from center1.
590 double c1_x = std::get<0>(center1) + std::sqrt(std::pow(radius, 2) - std::pow(std::get<1>(mid) - std::get<1>(center1), 2));
591 double c1_y = std::get<1>(center1) + (std::get<1>(mid)-std::get<1>(center1));
592
593 // Solution from center2.
594 double c2_x = std::get<0>(center2) + std::sqrt(std::pow(radius, 2) - std::pow(std::get<1>(mid) - std::get<1>(center2), 2));
595 double c2_y = std::get<1>(center2) + (std::get<1>(mid)-std::get<1>(center2));
596
597 if(curvature_sign >= 0)
598 {
599 alignHeadingToTangent(_handle, c1_x, c1_y);
600 _handle->setPose(c1_x, c1_y, _handle->time(), _handle->heading());
601 }
602 else
603 {
604 alignHeadingToTangent(_handle, c2_x, c2_y);
605 _handle->setPose(c2_x, c2_y, _handle->time(), _handle->heading());
606 }
607 }
608 }

610 void RRT::applyAbsoluteConstraints(Node* _handle, Node* _nearest)
611 {
612 double dist = calcDist(_handle, _nearest, true);
613 double angle = calcAngle(_handle, _nearest);
614 double tm = _handle->time();
615
618 if((std::abs(angle) > max_angle_rad_))
619 {
620 angle = ((angle / std::abs(angle)) * max_angle_rad_) + _nearest->heading();
621 }
622
623 if(std::abs(dist) > max_dist_)
624 {
625 dist = (dist / std::abs(dist)) * max_dist_;
626 }
627 else if (std::abs(dist) < min_dist_)
628 {
629 dist = (dist / std::abs(dist)) * min_dist_;
630 }
631
633 if(tm < _nearest->time())
634 {
635 tm = _nearest->time() + 0.0001F;
636 }
637
638 else if(std::abs(tm - _nearest->time()) > max_interval_)
639 {
640 tm = _nearest->time() + max_interval_;
641 }
642
643 double x = (dist * std::cos(angle)) + _nearest->xCrdnt();
644 double y = (dist * std::sin(angle)) + _nearest->yCrdnt();
645
647 _handle->setPose(x,y,tm, angle);
648 _handle->setBackEdgeWeight(dist); //Update this with Lat Acceleration
649 }

663 double RRT::calcLongAccel(Node* _handle, Node* _ref)
664 {
666 double dist1 = calcDist(_handle, _ref, false);
667 double dist2 = calcDist(_ref, _ref->BackCnnctn(), false);
668 double delta_time1 = _handle->time() - _ref->time();
669 double delta_time2 = _handle->time() - _ref->BackCnnctn()->time();
670
671 if (delta_time1 > 0 && delta_time2 > 0)
672 {
673 double velocity1 = dist1 / delta_time1;
674 double velocity2 = dist2 / delta_time2;
675 return (velocity1 - velocity2) / delta_time1;
676 }
677 else if (delta_time1 > 0 && delta_time2 <= 0)
678 {
679 double velocity1 = dist1 / delta_time1;
680 return velocity1 / delta_time1; // Assuming initial velocity is zero
681 }
682 else
683 {
684 return 0;

705 void RRT::constrainLongitudinal(Node* _handle)
706 {
707 double jerk = calcLongJerk(_handle, _handle->BackCnnctn());
708 double accel = calcLongAccel(_handle, _handle->BackCnnctn());
709 double dist = calcDist(_handle, _handle->BackCnnctn(), false);
710 double dist0 = calcDist(_handle->BackCnnctn(), _handle->BackCnnctn()->BackCnnctn(), false);
711 double v0 = dist0 / (_handle->BackCnnctn()->time() - _handle->BackCnnctn()->BackCnnctn()->time());
712 double accel0 = calcLongAccel(_handle->BackCnnctn(), _handle->BackCnnctn()->BackCnnctn());
713
714 double time = _handle->time() - _handle->BackCnnctn()->time(); // initialize
715
716 if(dim_3D_)
717 {
718 if(std::abs(jerk) > max_long_jerk_ && std::abs(accel) > max_long_accel_)
719 {
722 time = (v0 - std::sqrt(v0 * v0 - 2 * max_long_accel_ * dist)) / max_long_accel_;
723 _handle->setPose(_handle->xCrdnt(), _handle->yCrdnt(), _handle->BackCnnctn()->time() + time, _handle->heading());
724
726 jerk = calcLongJerk(_handle, _handle->BackCnnctn());
727 if(std::abs(jerk) > max_long_jerk_)
728 {
731 time = (std::sqrt(3 * max_long_jerk_ * max_long_jerk_ + 12 * accel0 * accel0) - 3 * accel0) / max_long_jerk_;
732 _handle->setPose(_handle->xCrdnt(), _handle->yCrdnt(), _handle->BackCnnctn()->time() + time, _handle->heading());
733 }
734 }
735 else if (std::abs(jerk) <= max_long_jerk_ && std::abs(accel) > max_long_accel_)
736 {
739 time = (v0 - std::sqrt(v0 * v0 - 2 * max_long_accel_ * dist)) / max_long_accel_;
740 _handle->setPose(_handle->xCrdnt(), _handle->yCrdnt(), _handle->BackCnnctn()->time() + time, _handle->heading());
741 }
742 else if (std::abs(jerk) > max_long_jerk_ && std::abs(accel) <= max_long_accel_)
743 {
746 time = (std::sqrt(3 * max_long_jerk_ * max_long_jerk_ + 12 * accel0 * accel0) - 3 * accel0) / max_long_jerk_;
747 _handle->setPose(_handle->xCrdnt(), _handle->yCrdnt(), _handle->BackCnnctn()->time() + time, _handle->heading());
748 }

752 bool RRT::isOccupied(Node *_handle)
753 {
754 static int sequential_check = 0;
755 bool occupied = false;
756
757 if (!occupancy_map_.has_value())
758 {
759 return false;
760 }
761
763 if (!_handle->BackCnnctn())
764 {
765 std::cerr << "Warning: Node has no back connection; cannot check edge occupancy. Terminating RRT" << std::endl;
766 cmplt = true; // Terminate RRT build
767 return true;
768 }
769
771 for (const auto &occupancy : occupancy_map_.value())
772 {
774 auto half_width = occupancy.second / 2.0F;
775 auto x_min = std::get<0>(occupancy.first) - half_width;
776 auto x_max = std::get<0>(occupancy.first) + half_width;
777 auto y_min = std::get<1>(occupancy.first) - half_width;
778 auto y_max = std::get<1>(occupancy.first) + half_width;
779 auto time_min = std::get<2>(occupancy.first);
780 auto time_max = std::get<2>(occupancy.first) + max_interval_;
781
783 if (_handle->xCrdnt() >= x_min && _handle->xCrdnt() <= x_max &&
784 _handle->yCrdnt() >= y_min && _handle->yCrdnt() <= y_max &&
785 _handle->time() >= time_min && _handle->time() <= time_max)
786 {
787 occupied = true;
788 std::cout << "Node in occupied space" << std::endl;
789 break;
790 }
791
792
795 auto line_x1 = _handle->BackCnnctn()->xCrdnt(); // Now safe
796 auto line_y1 = _handle->BackCnnctn()->yCrdnt();
797 auto line_z1 = _handle->BackCnnctn()->time();
798 auto line_x2 = _handle->xCrdnt();
799 auto line_y2 = _handle->yCrdnt();
800 auto line_z2 = _handle->time();
801
803 double t_min = 0.0;
804 double t_max = 1.0;
805 double dir_x = line_x2 - line_x1;
806 double dir_y = line_y2 - line_y1;
807 double dir_z = line_z2 - line_z1;
808
810 for (int i = 0; i < 3; i++) {
811 double min_val = (i == 0) ? x_min : (i == 1) ? y_min : time_min;
812 double max_val = (i == 0) ? x_max : (i == 1) ? y_max : time_max;
813 double p = (i == 0) ? line_x1 : (i == 1) ? line_y1 : line_z1;
814 double d = (i == 0) ? dir_x : (i == 1) ? dir_y : dir_z;
815
816 if (std::abs(d) < 1e-10) {
817 if (p < min_val || p > max_val) {
818 continue;
819 }
820 } else {
821 double t1 = (min_val - p) / d;
822 double t2 = (max_val - p) / d;
823 if (t1 > t2) {
824 std::swap(t1, t2);
825 }
826 t_min = std::max(t_min, t1);
827 t_max = std::min(t_max, t2);
828 if (t_min > t_max) {
829 continue;
830 }
831 }
832 }
834 if (t_min <= t_max && t_min <= 1.0 && t_max >= 0.0) {
835 occupied = true;
836 break;
837 }
838 }
839 if (occupied)
840 {
841 sequential_check++;
842 }
843 else
844 {
845 sequential_check = 0;
846 }
847
849 if (sequential_check > sequential_check_limit_)
850 {
851 std::cout << "Greater than " << sequential_check_limit_ << " Sequential Nodes in Occupied Space: possible non-admissible trajectory." << std::endl;
852
854 cmplt = true;
855 }
856 return occupied;
857 }

885 bool RRT::stepRRT()
886 {
887 pose_t output;
888 static auto i = 0;
889
890 if(cmplt)
891 {
892 std::cout << "RRT is already complete, no need to step." << std::endl;
893 return cmplt;
894 }
895
897 static double x_min = std::min(std::get<0>(range_a_), std::get<0>(range_b_));
898 static double x_max = std::max(std::get<0>(range_a_), std::get<0>(range_b_));
899 static double y_min = std::min(std::get<1>(range_a_), std::get<1>(range_b_));
900 static double y_max = std::max(std::get<1>(range_a_), std::get<1>(range_b_));
901
903 static std::random_device rand;
904 static std::mt19937 gen(rand());
905 static std::uniform_real_distribution<double> range_x(x_min, x_max);
906 static std::uniform_real_distribution<double> range_y(y_min, y_max);
907 static std::uniform_real_distribution<double> range_tm(0.0F, max_time_);
908
909 auto tm = range_tm(gen);
910 if (!dim_3D_)
911 {
912 tm = 0;
913 }
914
916 auto occupied = true;
917 while(occupied && !cmplt)
918 {
920 i++;
921
923 output = std::make_tuple(range_x(gen), range_y(gen), tm, 0.0F);
924
926 addNode(output, 0.0F);
927
930 applyConstraints(adjacencyList_.back());
931
933 if(isOccupied(adjacencyList_.back()))
934 {
936 deleteNode(adjacencyList_.back());
937 i--;
938 }
939 else
940 {
941 occupied = false;
942 }
943 }
944
946 checkDone();
947
948 if(i > iteration_limit_)
949 {
950 std::cout << "Iteration Limit of " << i-1 << " Reached, Stopping RRT. " << std::endl;
951 std::cout << "ITS POSSIBLE THERE IS NOT AN ADMISSIBLE SOLUTION" << std::endl;
952 cmplt = true;
953 }
954
955 if(cmplt && i <= iteration_limit_)
956 {
957 std::cout << "RRT Completed with: " << adjacencyList_.size() << " Nodes and " << i << " Iterations." << std::endl;
958 }
959
960 return cmplt;
961 }
